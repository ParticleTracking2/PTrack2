00128 void Chi2LibFFTW::conv2d_fft(MyMatrix<double> *img, MyMatrix<double> *kernel_img, MyMatrix<double> *output){
00129         MyLogger::log()->debug("[Chi2LibFFTW][conv2d_fft] Generating Convolution using FFTW");
00130         fftw_complex    *fft_image, *fft_kernel;
00131         fftw_plan       plan_forward_image, plan_forward_kernel, plan_backward;
00132         //auxiliary structures are necessary because fftw3 optimization plan will destroy it!
00133         double                  *ifft_result, *data, *kernel;
00134         int nwidth      =       (int)(img->sX()+kernel_img->sX()-1);
00135         int nheight     =       (int)(img->sY()+kernel_img->sY()-1);
00136 
00137         pthread_mutex_lock( &mutex1 );
00138         // FFTW Allocs
00139         size_t size = (size_t)(nwidth * nheight);
00140         //the new size includes zero padding space
00141         data            = fftw_alloc_real(size);
00142         kernel          = fftw_alloc_real(size);
00143         ifft_result = fftw_alloc_real(size);
00144 
00145         //fftw handle real fft avoiding redundancy in the complex plane, therefore the nheight/2
00146         size = (size_t)(nwidth*(floor(nheight/2) + 1));
00147         fft_image       = fftw_alloc_complex(size);
00148         fft_kernel      = fftw_alloc_complex(size);
00149 
00150         plan_forward_image      = fftw_plan_dft_r2c_2d( nwidth, nheight, data, fft_image, FFTW_ESTIMATE );
00151         plan_forward_kernel     = fftw_plan_dft_r2c_2d( nwidth, nheight, kernel, fft_kernel, FFTW_ESTIMATE );
00152         plan_backward           = fftw_plan_dft_c2r_2d( nwidth, nheight, fft_image, ifft_result, FFTW_ESTIMATE );
00153 
00154         pthread_mutex_unlock( &mutex1 );
00155 
00156         //populate kernel and shift input
00157         for(unsigned int x = 0 ; x < (unsigned int)nwidth ; ++x ){
00158                 unsigned int xnw = x*nwidth;
00159                 for(unsigned int y=0; y < (unsigned int)nheight; ++y){
00160                         if(x < kernel_img->sX() && y < kernel_img->sY())
00161                                 kernel[xnw+ y] = kernel_img->getValue(x,y);
00162                         else
00163                                 kernel[xnw+ y] = 0;
00164                 }
00165         }
00166 
00167         for(unsigned int x = 0 ; x < (unsigned int)nwidth ; ++x ){
00168                 unsigned int xnw = x*nwidth;
00169                 for(unsigned int y=0; y < (unsigned int)nheight; ++y){
00170                         if(x < img->sX() && y < img->sY())
00171                                 data[xnw+ y] = img->getValue(x,y);
00172                         else
00173                                 data[xnw+ y] = 0;
00174                 }
00175         }
00176 
00177         MyLogger::log()->debug("[Chi2LibFFTW][conv2d_fft] Starting FFTW");
00179                 //fft of image
00180                 fftw_execute( plan_forward_image );
00181                 //fft of kernel
00182                 fftw_execute( plan_forward_kernel );
00183 
00184                 //convolution in fourier domain
00185                 double f1, f2;
00186                 double nwnh = (double)(nwidth*nheight);
00187                 unsigned int limit = (unsigned int)(nwidth * (floor(nheight/2) + 1));
00188                 for(unsigned int i=0; i< limit; ++i){
00189                         f1 = fft_image[i][0]*fft_kernel[i][0] - fft_image[i][1]*fft_kernel[i][1];
00190                         f2 = fft_image[i][0]*fft_kernel[i][1] + fft_image[i][1]*fft_kernel[i][0];
00191 
00192                         fft_image[i][0]=f1/nwnh;
00193                         fft_image[i][1]=f2/nwnh;
00194                 }
00195 
00196                 //ifft of the product
00197                 fftw_execute( plan_backward );
00199         MyLogger::log()->debug("[Chi2LibFFTW][conv2d_fft] FFTW Finished");
00200 
00201         if(output->sX() == (unsigned int)nwidth && output->sY() == (unsigned int)nheight)
00202         for(unsigned int x = 0 ; x < output->sX() ; ++x ){
00203                 unsigned int xnw = x*nwidth;
00204                 for(unsigned int y = 0 ; y < output->sY() ; ++y ){
00205                         output->at(x,y) = ifft_result[xnw+y];
00206                 }
00207         }
00208 
00209     /* free memory */
00210     fftw_destroy_plan( plan_forward_image );
00211     fftw_destroy_plan( plan_forward_kernel );
00212     fftw_destroy_plan( plan_backward );
00213 
00214     fftw_free( data );
00215     fftw_free( kernel );
00216     fftw_free( ifft_result );
00217 
00218     fftw_free( fft_image );
00219     fftw_free( fft_kernel );
00220 }
