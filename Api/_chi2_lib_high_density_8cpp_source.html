00074 pair<double,double> Chi2LibHighDensity::gaussianFit(vector<MyPeak> *peaks, MyMatrix<double> *img, unsigned int ss){
00075         unsigned int slots = 21;
00076 
00077         MyLogger::log()->debug("[Chi2LibHighDensity][gassianFit] Calculating MU and SIGMA ");
00078         vector<double> X(slots);
00079         vector<int> freq(slots);
00080         double dx = 0.01;
00081         for(unsigned int i=0; i < slots; ++i){
00082                 freq[i]=0;
00083                 X[i]=0.8+i*dx;
00084         }
00085 
00086         vector<MyPeak> new_peaks;
00087         new_peaks.reserve(peaks->size());
00088         MyPeak tmp;
00089         for(unsigned int i=0; i < peaks->size(); ++i){
00090                         tmp.x = peaks->at(i).x - ss;
00091                         tmp.y = peaks->at(i).y - ss;
00092                         tmp.intensity = img->getValue(tmp.x, tmp.y);
00093                         new_peaks.push_back(tmp);
00094         }
00095 
00096         sort(new_peaks.begin(), new_peaks.end(), MyPeak::compareMe);
00097 
00098         unsigned int current = 0;
00099         for(unsigned int i=0; i < new_peaks.size(); ++i){
00100                 for(unsigned int s=current; s < slots; ++s){
00101                         if(new_peaks.at(i).intensity <= X[s] + dx){
00102                                 ++freq[s];
00103                                 break;
00104                         }
00105                         ++current;
00106                 }
00107         }
00108 
00109         int check = 0;
00110         double mu = 0.0, sigma = 0.0;
00111         for(unsigned int i=0; i < slots; ++i){
00112                 mu += X[i]*freq[i];
00113                 check += freq[i];
00114         }
00115         mu = mu/check;
00116 
00117         for(unsigned int i=0; i < slots; ++i){
00118                 sigma += freq[i]*(mu-X[i])*(mu-X[i]);
00119         }
00120         sigma = sqrt(sigma/check);
00121 
00122         MyLogger::log()->debug("[Chi2LibHighDensity][gassianFit] Total Checks=%i; Returning MU=%f; SIGMA=%f", check, mu, sigma);
00123         pair<double, double> ret; ret.first = mu; ret.second = sigma;
00124         return ret;
